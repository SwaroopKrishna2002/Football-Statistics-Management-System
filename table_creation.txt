CREATE TABLE LEAGUE(League_id VARCHAR(10),LName VARCHAR(30),Country VARCHAR(20),Association VARCHAR(30),Division VARCHAR(5),PRIMARY KEY(League_id));
CREATE TABLE CLUB(League_id VARCHAR(10),Club_Name VARCHAR(40),Ranking VARCHAR(10),Stadium VARCHAR(40),Sponsor VARCHAR(10),Club_Value VARCHAR(15),PRIMARY KEY(Club_Name),FOREIGN KEY(League_id) REFERENCES LEAGUE(League_id));
CREATE TABLE MANAGER(Club_Name VARCHAR(40),Mgr_Name VARCHAR(30),Mgr_id VARCHAR(10),Matches_Managed NUMERIC(4),Won NUMERIC(3),Lost NUMERIC(3),Draw NUMERIC(3),Goals_scored NUMERIC(4),Goals_conceded NUMERIC(3),League_pos NUMERIC(2),PRIMARY KEY(Mgr_id),FOREIGN KEY(Club_Name) REFERENCES CLUB(Club_Name));
CREATE TABLE PLAYER(Player_id VARCHAR(10),Club_Name VARCHAR(40),Nationality VARCHAR(30),Position VARCHAR(25),Age NUMERIC(3),Market_Value VARCHAR(15),Mgr_id VARCHAR(10),PRIMARY KEY(Player_id),FOREIGN KEY(Club_Name) REFERENCES CLUB(Club_Name),FOREIGN KEY(Mgr_id) REFERENCES MANAGER(Mgr_id));
CREATE TABLE OUTFIELD_PLAYER(Player_id VARCHAR(10),Appearances NUMERIC(4),Outfield_pos VARCHAR(25),Fouls NUMERIC(3),Yellow_Cards NUMERIC(3),Red_Cards NUMERIC(2),FOREIGN KEY(Player_id) REFERENCES PLAYER(Player_id));
CREATE TABLE GOALKEEPER(Player_id VARCHAR(10),Appearances NUMERIC(4),Clean_sheets NUMERIC(3),Saves NUMERIC(3),Penalties_Saved NUMERIC(2),Conceded_goals NUMERIC(3),Fouls NUMERIC(3),Yellow_Cards NUMERIC(3),Red_Cards NUMERIC(2),FOREIGN KEY(Player_id) REFERENCES PLAYER(Player_id));
CREATE TABLE ATTACKING_STATS(Outfield_pos VARCHAR(25),Goals NUMERIC(3),Shots NUMERIC(3),xG NUMERIC(3,2),Offsides NUMERIC(3),Touches_Opp_Box NUMERIC(4),Player_id VARCHAR(15),FOREIGN KEY(Player_id) REFERENCES player(Player_id));
CREATE TABLE TEAMPLAY(Outfield_pos VARCHAR(25),Passes NUMERIC(4),Passes_completed NUMERIC(4),Passing_accuracy NUMERIC(3),Crosses NUMERIC(3),Dribbles NUMERIC(3),Duels_won NUMERIC(3),Assists NUMERIC(3),Player_id VARCHAR(15),FOREIGN KEY(Player_id) REFERENCES player(Player_id));
CREATE TABLE DEFENSIVE_STATS(Outfield_pos VARCHAR(25),Tackles NUMERIC(3),Blocks NUMERIC(3),Aerial_duels NUMERIC(3),Interceptions NUMERIC(3),Own_goals NUMERIC(2),Player_id VARCHAR(15),FOREIGN KEY(Player_id) REFERENCES player(Player_id));
